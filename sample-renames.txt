rename to add leading 0 to sample nums. cd 







find . -type f -regextype posix-extended -regex './23908X[0-9]_[^/]' | while read FILE; do
  # Extract the base name of the file
  BASENAME=$(basename "$FILE")
  # Construct the new name by inserting a 0 before the single digit
  NEWNAME=$(echo "$BASENAME" | sed -E 's/(23908X)([0-9])_/\10\2_/')
  # Rename the file
  mv "$FILE" "$(dirname "$FILE")/$NEWNAME"
done
